CREATE TABLE ATTORI (
CodAttore Integer NOT NULL PRIMARY KEY,
Nome VARCHAR(30) NOT NULL,  
Nazionalità VARCHAR(50),
AnnoNascita date
);

CREATE TABLE FILM (
CodFilm Integer PRIMARY KEY,

Titolo VARCHAR(30),
AnnoProduzione INT,
Nazionalità VARCHAR(30),
Regista VARCHAR(30),
Genere VARCHAR(30)
);
CREATE TABLE RECITA (
CodAttore Integer ,
CodFilm Integer ,
    FOREIGN KEY (CodAttore) REFERENCES ATTORI(CodAttore),
     FOREIGN KEY (CodFilm) REFERENCES FILM(CodFilm)
);

CREATE TABLE SALE (
CodSala Integer PRIMARY KEY,
Posti Integer,
Nome VARCHAR(30),
Città VARCHAR(30)
);

CREATE TABLE PROIEZIONI(
CodProiezione Integer PRIMARY KEY,
CodFilm Integer ,
CodSala Integer ,
Incasso Integer,
    DataProiezione date,
    FOREIGN KEY (CodFilm) REFERENCES FILM(CodFilm),
    FOREIGN KEY (CodSala) REFERENCES SALE(CodSala)
);
